\question Determine whether the following series are convergent or divergent. Show all work to recieve full credit. An answer of convergent or divergent alone will yeild no credit.
\begin{parts}
    \part[10] $\displaystyle\sum_{n=2}^\infty\frac{\ln n}{n^3}$
    \begin{solution}[\stretch{1}]
        Let $\displaystyle b_n=\frac{n}{n^3}\ge\frac{\ln n}{n^3}=a_n$ for all $n$. Then $\displaystyle\sum b_n=\sum\frac{1}{n^2}$ which is a convergent $p$-series with $p=2>1$. Therefore, $\displaystyle\sum a_n$ is convergent by the direct comparison test.
    \end{solution}

    \part[10]$\displaystyle\sum_{i=1}^\infty \left(\sqrt[i]{i}-1\right)^i$
    \begin{solution}[\stretch{1}]
        Using the root test: \[\lim_{i\to\infty}\sqrt[i]{\left(\sqrt[i]{i}-1\right)^i}=\lim_{i\to\infty}\sqrt[i]{i}-1=1-1=0<1.\]
        Therefore (absolutely) convergent by the root test.
    \end{solution}
    

    \part[10]$\displaystyle\sum_{n=0}^\infty (-1)^{n+1} e^{\left(\frac{2}{n+1}\right)}$
    \begin{solution}[\stretch{1}]
        \[lim_{n\to\infty}e^{\frac{2}{n+1}}=e^0=1.\] Therefore divergent by the divergence test.\phantom{asdljfh}
    \end{solution}

    \part[10]$\displaystyle\sum_{k=1}^\infty \frac{3\sqrt{k}+2k}{2k^{1.5}+k^2}$
    \begin{solution}[\stretch{1}]
        If $\displaystyle f(x)=\frac{3\sqrt{x}+2x}{2x^{1.5}+x^2}$, then $f$ is positive, continuous and decreasing. Now, \[\int_1^\infty\frac{3\sqrt{x}+2x}{2x^{1.5}+x^2}=\lim_{b\to\infty}\left[\ln\left|2b^{1.5}+b^2\right|-\ln(3)\right]=\infty\therefore\text{Divegent}.\] Thus $\sum a_n$ is divergent by the integral test.
    \end{solution}

\end{parts}
